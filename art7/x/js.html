<script>
     let area;
     let circleDiameter;
     let circleRadius = Number(prompt("Який радіус кола?"));

     circleDiameter = 2 * circleRadius;
     area = 3.14 * (circleDiameter ** 2) / 4;

     alert(circleDiameter);
     alert("Площа кола = " + area.toFixed(1) + " кв.cм");
</script>
